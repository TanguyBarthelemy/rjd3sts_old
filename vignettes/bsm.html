a<-sts(log(retail$RetailSalesTotal), seasonal = "HarrisonStevens")
cat("\nBSM with Harrison-Stevens seasonal component")
#> 
#> BSM with Harrison-Stevens seasonal component
print(a)
#> Structural time series 
#> 
#> Variances:
#> level:  0.000096 
#> slope:  0 
#> seasonal:  0.000002 
#> noise:  0.000188 
#> 
#> LogLikelihood:  500.4679 
#> AIC:  -990.9358 
#> 
#> No regression variable
level<-a$decomposition$level$data
elevel<-a$decomposition$level$stde
plot(level, type='l')
lines(level+2*elevel, col='red')
lines(level-2*elevel, col='red')

seas<-a$decomposition$seasonal$data
eseas<-a$decomposition$seasonal$stde
plot(seas, type='l')
lines(seas+2*eseas, col='red')
lines(seas-2*eseas, col='red')
